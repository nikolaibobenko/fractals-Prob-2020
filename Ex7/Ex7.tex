We call a set $H \subset \mathbb{N}$ Poincaré if $\forall S \subset \mathbb{N}$ with $\bar{d}(S) > 0$ we have
\[(S-S) \cap H \neq \emptyset\]

\begin{exercise}
    Let $H \subset \mathbb{N}$ be finite. We denote by 
    \[ K_H = \{x = \sum x_n 2^{-n} :\; x_n x_{n+h} = 0 \text{ if } h \in H\}. \]
    Show that $\dim(K_H) = \dim_M(K_H)$. How can we calculate $\dim(K_H)$?
\end{exercise}

\begin{exercise}
    For the following sets check whether they are Poincaré or not:
    \begin{enumerate}
        \item $k \mathbb{N}$
        \item $2 \mathbb{N} + 1$
        \item $\{n^2\}_n$
        \item $\{n^2 + 1\}_n$
    \end{enumerate}
\end{exercise}
